@startuml
actor User
participant StatsController
participant UserController
participant StatsRepository
participant Database

User -> StatsController: PUT /estatisticas_jogador/{id} (id, points, assists, rebounds, steals, blocks, three_pointers, two_pointers, free_throws, Token)
activate StatsController

StatsController -> UserController: validateToken(Token)
activate UserController

alt Token válido
    UserController --> StatsController: Token válido
    deactivate UserController

    StatsController -> StatsRepository: updatePlayerStats(id, points, assists, rebounds, steals, blocks, three_pointers, two_pointers, free_throws)
    activate StatsRepository

    StatsRepository -> Database: updatePlayerStats(id, points, assists, rebounds, steals, blocks, three_pointers, two_pointers, free_throws)
    activate Database

    Database --> StatsRepository: Confirmação
    deactivate Database

    StatsRepository --> StatsController: Confirmação
    deactivate StatsRepository

    StatsController --> User: Resposta sucesso (status, PlayerStats)
    deactivate StatsController
else Token inválido
    UserController --> StatsController: Token inválido
    deactivate UserController

    StatsController --> User: Resposta erro (401 Unauthorized)
    deactivate StatsController
end
@enduml
